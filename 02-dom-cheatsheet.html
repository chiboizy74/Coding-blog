DOM Cheatsheet: querySelector by example

Select elements

document.querySelector('#id')      // first match by id
document.querySelector('.btn')     // first with class
document.querySelector('button')   // first <button>
document.querySelectorAll('li')    // NodeList of every <li>

Iterate

document.querySelectorAll('li').forEach(el => {
  el.classList.add('highlight');
});

Edit content & attributes

const img = document.querySelector('img.thumb');
img.alt = 'Thumbnail';             // attribute
img.setAttribute('width', '320');  // attribute via setter
img.src = '/path/to/image.jpg';    // property

Events

document.querySelector('button.save').addEventListener('click', e => {
  e.preventDefault();
  console.log('Saved!');
});
